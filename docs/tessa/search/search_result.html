 10# ----- The predicates used to sort and bucketize results based on a query -----

 12# Each matches_* function takes a `query` and a `Symbol` and returns an int denoting the
 13# match, where lower numbers signify better matches but 0 (=falsy) signifies not match.
 14
 15
 16def matches_entire_name_or_alias(query: str, symbol: Symbol) -> int:
 17    """Try to match the complete name or one of the aliases completely. E.g.:
 18    `matches_entire_name_or_alias("eth", Symbol("ETH")) > 0`.
 19    """
 20    query = query.lower()
 21    if query == symbol.name.lower():
 22        return 1
 23    if any(query == x.lower() for x in symbol.aliases if x):
 24        return 2
 25    return 0
 26
 27
 28def matches_word_boundary(query: str, symbol: Symbol) -> int:
 29    """Try to match a complete word in name, aliases, or query. E.g.:
 30    `matches_word_boundary("eth", Symbol("ETHW", aliases=["PoW ETH"])) > 0`.
 31    """
 32    query = query.lower()
 33    pattern = re.compile(rf"(^|[^a-z0-9]){query}([^a-z0-9]|$)", re.IGNORECASE)
 34    if re.search(pattern, symbol.name):
 35        return 1
 36    if re.search(pattern, "|".join(symbol.aliases)):
 37        return 2
 38    if re.search(pattern, str(symbol.query)):
 39        return 3
 40    return 0
 41
 42
 43def matches_inanyway(query: str, symbol: Symbol) -> int:
 44    """Try to match anywhere in name, aliases, or query. E.g.:
 45    `matches_inanyway("eth", Symbol("TOGETHER")) > 0`.
 46    """
 47    query = query.lower()
 48    if query in symbol.name.lower():
 49        return 1
 50    if query in ("|".join(symbol.aliases)).lower():
 51        return 2
 52    if query in str(symbol.query).lower():
 53        return 3
 54    return 0

 57# ----- Everything related to bucketizing symbols -----
 58
 59Bucket = NamedTuple("Bucket", [("name", str), ("symbols", List[Symbol])])
 60"""A bucket is a set of results of a certain quality level in a larger search result. A
 61bucket has a name and a list of symbols that belong into that bucket.
 62"""
 63
 64
 65BUCKET_SPEC = [
 66    ("Matches entire name or alias ⭐", matches_entire_name_or_alias),
 67    ("Matches word boundary", matches_word_boundary),
 68    ("Matches in any way", matches_inanyway),
 69    # A bucket with the non-matches will be added by `bucketize`.
 70]
 71
 72
 73def bucketize(query: str, symbols: List[Symbol]) -> List[Bucket]:
 74    """Split a list of `Symbol`s into buckets based on how well a symbol matches the
 75    `query`.
 76
 77    (We're not using `itertools.groupby` here bc we need empty lists for empty buckets,
 78    which `groupby` doesn't provide.)
 79    """
 80    buckets = []
 81    rest = symbols
 82    for name, matches_predicate in BUCKET_SPEC:
 83        matches_query_predicate = functools.partial(matches_predicate, query)
 84        hits = list(filter(matches_query_predicate, rest))
 85        rest = list(itertools.filterfalse(matches_query_predicate, rest))
 86        buckets.append(Bucket(name, hits))
 87    buckets.append(Bucket("Matches not (but was somehow still returned) 🤔", rest))
 88    return buckets

 94def create_sort_key_for_query(query: str) -> Callable:
 95    """Create a sort key function that classifies a `Symbol` based on `query`."""
 96
 97    def symbol_sort_key(symbol: Symbol) -> int:
 98        """Cycle through the predicates in `BUCKET_SPEC` and use an offset to return an
 99        int based on the matching quality.
100        """
101        offset = 10
102        for _, pred in BUCKET_SPEC:
103            score = pred(query, symbol)
104            if score > 0:
105                return offset + score
106            offset += 10
107        return 999
108
109    return symbol_sort_key

112def remove_duplicates(symbols: List[Symbol]) -> List[Symbol]:
113    """Return a list in which every symbol is unique. Two symbols are considered
114    equivalent if they match in `query` and `source`. Regardless of whether their names
115    are the same or different.
116
117    Note further that several symbols can have the same name but still be considered
118    different if they differ in any of `query` or `source`.
119    """
120    equality_key = lambda symbol: str((symbol.source, symbol.query))
121    return [k for k, _ in itertools.groupby(sorted(symbols, key=equality_key))]

127class SearchResult:
128    """Manages a search result consisting of a list of `Symbol`s. Removes duplicates,
129    bucketizes, sorts and offers methods to filter and print statistics.
130
131    (Design note 1: `SearchResult` has no relationship with `SymbolCollection` because
132    the purposes of the two classes are very different and because they have different
133    equality definitions for symbols.
134
135    Design note 2: `SearchResult` and `SymbolCollection` have different definitions of
136    equality between symbols: `SearchResult` considers 2 symbols to be equal if they are
137    equal in `query` and `source`, regardless of whether they have the same name or not.
138    A `SymbolCollection` does not have a definition of equality, but it does enforce
139    that names are unique.)
140
141    Example use:
142
143    ```python
144    from tessa import search
145    r = search("harmony")
146
147    # Review results:
148    r.p()
149
150    # Get the 1 symbol from source "coingecko" in the best bucket (i.e., bucket 0):
151    s = r.filter(source="coingecko").buckets[0].symbols[0]
152    s.price_latest()
153    ```
154    """
155
156    query: str
157    """The query that produced the results here."""
158
159    symbols: List[Symbol]
160    """List of symbols in this result. Always deduplicated and sorted."""
161
162    buckets: List[Bucket]
163    """List of buckets after bucketizing the symbols."""
164
165    filter_history: List[str]
166    """The filters that have been applied to get to this result."""
167
168    def __init__(self, query: str, symbols: List[Symbol]):
169        self.query = query
170        self.buckets = []
171        self.filter_history = []
172        self.symbols = []
173        self.add_symbols(symbols)
174
175    def add_symbols(self, symbols: List[Symbol]) -> SearchResult:
176        """Add symbols. Removes duplicates, sorts, and bucketizes after adding."""
177        self.symbols.extend(symbols)
178        self.symbols = remove_duplicates(self.symbols)
179        self.symbols.sort(key=create_sort_key_for_query(self.query))
180        self.buckets = bucketize(self.query, self.symbols)
181        return self
182
183    def filter(self, **kwargs) -> SearchResult:
184        """Filter for arbitrary attribute-value-pairs (e.g., `source="coingecko"` or
185        `exchange="ebs"`) and return a new `SearchResult` with the results after the
186        filtering. Also updates the filter history.
187        """
188        symbols = []
189        for s in self.symbols:
190            if all(getattr(s, k, "DOESNOTEXISTEVER") == v for k, v in kwargs.items()):
191                symbols.append(s)
192        newres = SearchResult(self.query, symbols)
193        newres.filter_history = self.filter_history + [
194            f"{k}={v}" for k, v in kwargs.items()
195        ]
196        return newres
197
198    def __str__(self) -> str:
199        """Return summary."""
200        out = ""
201
202        def s(line: str = "", indent=False) -> None:
203            nonlocal out
204            indent_str = "  " if indent else ""
205            out += indent_str + line + "\n"
206
207        s(f"Search results for query '{self.query}'")
208        if len(self.filter_history) > 0:
209            s("With filters: " + ", ".join(self.filter_history))
210        s()
211        for i, b in enumerate(self.buckets):
212            s(f"Bucket {i}: {b.name}")
213            sources = set(s.source for s in b.symbols)
214
215            if len(b.symbols) == 0:
216                s("No hits", indent=True)
217            else:
218                s(
219                    f"{len(b.symbols)} hits, {len(sources)} sources",
220                    indent=True,
221                )
222                if len(b.symbols) in range(1, 6):
223                    s("Hits: " + ", ".join([s.name for s in b.symbols]), indent=True)
224                if len(sources) in range(1, 6):
225                    s("Sources: " + ", ".join(sources), indent=True)
226            s()
227        return out
228
229    def p(self) -> None:
230        """Convenience method to print str rep."""
231        print(self)
